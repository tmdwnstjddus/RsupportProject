CREATE TABLE NOTICE     // NOTICE 테이블 생성
(
	UPLOADNO NUMBER NOT NULL PRIMARY KEY, // 게시물 번호 컬럼
	TITLE VARCHAR2 (100) NOT NULL, // 제목 컬럼
	UPLOADER VARCHAR2 (20) NOT NULL, // 작성자 컬럼
	CONTENT VARCHAR2 (1000), // 내용 컬럼
	DELETED CHAR (1) DEFAULT ('0') // 삭제 컬럼
	REGDATE DATE DEFAULT (SYSDATE), // 작성일
	MODDATE DATE DEFAULT (SYSDATE), // 최종수정일
	READCOUNT NUMBER DEFAULT (0) //조회수

);
CREATE SEQUENCE NOTICE _SEQUENCE; // NOTICE 시퀀스 생성


CREATE TABLE NOTICEMEMBER // NOTICEMEMBER 테이블 생성
(
	MEMBERID VARCHAR2 (1000) NOT NULL, //아이디
	PASSWD VARCHAR2 (1000) NOT NULL, //비밀번호
	EMAIL VARCHAR2 (2000) NOT NULL //이메일
);
